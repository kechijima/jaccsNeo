```
├── useCustomers.ts      # 顧客データ取得・CRUD

├── useSpaces.ts         # スペース・投稿管理

├── useNotifications.ts  # 通知リアルタイム取得
├── useCsvImport.ts      # CSVインポート処理（システム管理者・EM2以上）
├── useCsvExport.ts      # CSVエクスポート処理（担当顧客のみ・全ロール共通）
└── usePermission.ts     # ロール・権限チェックユーティリティ
                         # ロール: system_admin / board / em2_above / general / specialist
                         # 専門チームフラグ: specialTeams: string[] (例: ["real_estate", "non_life"])
```

## 8. モバイル対応方針

| ポイント | 対応方法 |
|---|---|
| ナビゲーション | PC: サイドバー / SP: ボトムナビ（Tailwindのmd:ブレークポイントで切替） |
| 顧客一覧 | SPでは横スクロールなし・カード表示に切替 |
| ダッシュボード | グラフはSPで縦並びに変更 |
| 投稿フォーム | SPでフルスクリーンモーダル |
| テーブル | 横スクロール or カード形式に切替 |
| PWA | nuxt/pwa モジュールでオフラインキャッシュ対応 |
